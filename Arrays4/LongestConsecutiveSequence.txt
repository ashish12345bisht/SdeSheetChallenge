class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        set<int> s;
        int n=nums.size();
        if(n==0)return 0;
    for(int i=0;i<n;i++){
        s.insert(nums[i]);
    }
    int ans=INT_MIN;
    for(int i=0;i<n;i++){
        if(!s.count(nums[i]-1)){
            int t=0;
            int x=nums[i];
            while(s.count(x)){
                x++;
                t++;
            }
            ans=max(ans,t);
        }
    }
    return ans;
    }
};
